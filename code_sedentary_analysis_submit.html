def __replace_header(self):
    metadata = json.load(open(self.__dataset_name + ".json"))
    title = metadata["selectedAttributes"]
    title = {i["name"]: i["title"] for i in title}
    self.data.columns = [self.__pretty(title[i]) for i in self.data.columns]
    self.data.index.name = title[self.data.index.name]
